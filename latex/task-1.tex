% ================= SECTION A =================
\section*{\textbf{SECTION A}}
\setcounter{section}{1}

\begin{enumerate}[label=\thesection.\arabic*, leftmargin=*]

\item Find the value of $k$ for which the roots of the quadratic equation  
$(k-5)x^2 + 2(k-5)x + 2 = 0$ are equal.

\item Find the value of $y$ for which the distance between the points  
$(2,-3)$ and $(10,y)$ is $10$ units.

\item Write whether the rational number $\frac{13}{3125}$ has a decimal
expansion which is terminating or non-terminating repeating.

\item Write the $n^{th}$ term of the A.P.  
$\frac{1}{k}, \frac{1+k}{k}, \frac{1+2k}{k}, \ldots$

\item If $\sin\theta + \cos\theta = \sqrt{2}\cos(90^\circ-\theta)$,
find the value of $\cot\theta$.

\item DE is drawn parallel to the base BC of $\triangle ABC$, meeting AB at D
and AC at E. If $\frac{AB}{CD}=4$ and $CE=2$ cm, find $AE$.

\end{enumerate}

% ================= SECTION B =================
\section*{\textbf{SECTION B}}
\setcounter{section}{2}

\begin{enumerate}[label=\thesection.\arabic*, leftmargin=*]

\item A bag contains 5 red balls and some blue balls.
If the probability of drawing a blue ball is three times that of drawing a red
ball, find the number of blue balls.

\item The $5^{th}$ and $15^{th}$ terms of an A.P. are 13 and $-17$
respectively. Find the sum of the first 21 terms.

\item Using Euclid’s Division Algorithm, find the HCF of 225 and 867.

\item If the point $(0,2)$ is equidistant from $(3,k)$ and $(k,5)$,
find the value of $k$.

\item Find the value of $a$ for which the pair of linear equations  
$2x+3y=7$ and $4x+ay=14$ has infinitely many solutions.

\end{enumerate}

% ================= SECTION C =================
\section*{\textbf{SECTION C}}
\setcounter{section}{3}

\begin{enumerate}[label=\thesection.\arabic*, leftmargin=*]

\item Show that any positive odd integer is of the form $4q+1$ or $4q+3$
for some integer $q$.

\item The ten’s digit of a number is twice its unit’s digit.
The number obtained by interchanging the digits is 36 less than the original
number. Find the original number.

\item The line segment joining $A(2,1)$ and $B(5,-8)$ is trisected at points
$P$ and $Q$. If $P$ lies on the line $2x-y+k=0$, find the value of $k$.

\item Show that $1$, $\frac{1}{2}$ and $-2$ are the zeroes of the polynomial  
$2x^3 + x^2 - 5x + 2$.

\item Prove that the angle between the two tangents drawn from an external
point to a circle is supplementary to the angle subtended by the line segment
joining the points of contact at the centre.

\end{enumerate}
