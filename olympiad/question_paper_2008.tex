% ================= PROBLEM 1 =================
\textbf{Problem 1.}
An acute-angled triangle $ABC$ has orthocentre $H$.
The circle passing through $H$ with centre the midpoint of $BC$ intersects
the line $BC$ at $A_1$ and $A_2$.
Similarly define $B_1,B_2,C_1,C_2$.
Show that $A_1,A_2,B_1,B_2,C_1,C_2$ lie on a circle.

\vspace{0.4cm}

% ================= PROBLEM 2 =================
\textbf{Problem 2.}
\begin{enumerate}
\item[(a)] Prove that
\[
\frac{x^2}{(x-1)^2} + \frac{y^2}{(y-1)^2} + \frac{z^2}{(z-1)^2} \ge 1
\]
for all real numbers $x,y,z\neq1$ with $xyz=1$.

\item[(b)] Prove that equality holds for infinitely many rational triples.
\end{enumerate}

\vspace{0.4cm}

% ================= PROBLEM 3 =================
\textbf{Problem 3.}
Prove that there exist infinitely many positive integers $n$ such that
$n^2 + 1$ has a prime divisor greater than $2n + \sqrt{2n}$.

% ================= PROBLEM 4 =================
\textbf{Problem 4.}
Find all functions $f:(0,\infty)\to(0,\infty)$ such that
\[
\frac{(f(w))^2 + (f(x))^2}{f(y^2) + f(z^2)} =
\frac{w^2 + x^2}{y^2 + z^2}
\]
for all positive real numbers $w,x,y,z$ satisfying $wx=yz$.

\vspace{0.4cm}

% ================= PROBLEM 5 =================
\textbf{Problem 5.}
Let $n$ and $k$ be positive integers with $k \ge n$ and $k-n$ even.
Let $2n$ lamps be given, each either on or off.
Initially all lamps are off.
At each step exactly one lamp is switched.

Let $N$ be the number of sequences of $k$ steps resulting in lamps
$1$ through $n$ on and lamps $n+1$ through $2n$ off.

Let $M$ be the number of such sequences where lamps $n+1$ through $2n$
are never switched on.
Determine the ratio $N/M$.

\vspace{0.4cm}

% ================= PROBLEM 6 =================
\textbf{Problem 6.}
Let $ABCD$ be a convex quadrilateral with $|BA|\neq|BC|$.
Denote the incircles of triangles $ABC$ and $ADC$ by $\omega_1$ and $\omega_2$.
Prove that the common external tangents of $\omega_1$ and $\omega_2$
intersect on a circle tangent to the four given lines.
