\begin{enumerate}

\item \textbf{Problem 1.}  
Real numbers $a_1, a_2, \ldots, a_n$ are given. For each $i$ $(1 \le i \le n)$ define
\[
d_i = \max\{a_j : 1 \le j \le i\} - \min\{a_j : i \le j \le n\}
\]
and let
\[
d = \max\{d_i : 1 \le i \le n\}.
\]

(a) Prove that for any real numbers $x_1 \le x_2 \le \cdots \le x_n$,
\[
\max\{|x_i - a_i| : 1 \le i \le n\} \ge \frac{d}{2}.
\]

(b) Show that there exist real numbers $x_1 \le x_2 \le \cdots \le x_n$ such that equality holds.

\item \textbf{Problem 2.}  
Consider five points $A,B,C,D,E$ such that $ABCD$ is a parallelogram and $BCED$ is a cyclic quadrilateral. Let $\ell$ be a line passing through $A$. Suppose that $\ell$ intersects the interior of segment $DC$ at $F$ and intersects line $BC$ at $G$. Suppose also that $EF = EG = EC$. Prove that $\ell$ is the bisector of angle $DAB$.

\item \textbf{Problem 3.}  
In a mathematical competition some competitors are friends. Friendship is mutual. A group of competitors is called a \textit{clique} if each two of them are friends. Given that the largest size of a clique is even, prove that the competitors can be arranged into two rooms such that the largest clique in each room has the same size.

\end{enumerate}

\item \textbf{Problem 4.}  
In triangle $ABC$ the bisector of angle $BCA$ intersects the circumcircle again at $R$. The perpendicular bisectors of $BC$ and $AC$ meet at $P$ and $Q$ respectively. Let $K$ and $L$ be the midpoints of $BC$ and $AC$. Prove that triangles $RPK$ and $RQL$ have the same area.

\item \textbf{Problem 5.}  
Let $a$ and $b$ be positive integers. Show that if
\[
4ab - 1 \mid (4a^2 - 1)^2,
\]
then $a = b$.

\item \textbf{Problem 6.}  
Let $n$ be a positive integer and consider
\[
S = \{(x,y,z) : x,y,z \in \{0,1,\ldots,n\},\ x+y+z > 0\}.
\]
Determine the smallest number of planes whose union contains $S$ but does not contain $(0,0,0)$.

\end{enumerate}
